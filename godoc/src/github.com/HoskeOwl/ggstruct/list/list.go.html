    71  		}

    73  	}

   128  // PushFront Add value to the start of the list.
   129  func (l *List[T]) PushFront(values ...T) {
   130  	for i := len(values) - 1; i >= 0; i-- {
   131  		l.len += 1
   132  		var newNode *element[T]
   133  		if l.root != nil {
   134  			newNode = &element[T]{data: values[i], next: l.root, prev: l.root.prev}
   135  			newNode.prev.next = newNode
   136  			l.root.prev = newNode
   137  		} else {
   138  			newNode = &element[T]{data: values[i]}
   139  			newNode.next = newNode
   140  			newNode.prev = newNode
   141  		}
   142  		l.root = newNode
   143  	}
   144  	return
   145  }

   163  	}

   167  // AddAfterIndex Add value to the specific position.
   168  // If list is 1 <-> 2 <-> 3 <-> 4 then you call this function with index 2 (for example with value 9).
   169  // You will get 1 <-> 2 <-> 3 <-> 9 <-> 4
   170  func (l *List[T]) AddAfterIndex(index int, values ...T) bool {
   171  	if l.root == nil || index < 0 || l.len <= index {
   172  		return false
   173  	}
   174  
   175  	var newNode *element[T]
   176  	// go to index node
   177  	current := l.root
   178  	for i := 0; i < index; i++ {
   179  		current = current.next
   180  	}
   181  
   182  	for _, v := range values {
   183  		l.len += 1
   184  		newNode = &element[T]{data: v}
   185  		newNode.next = current.next
   186  		newNode.prev = current
   187  		current.next = newNode
   188  		newNode.next.prev = newNode
   189  		current = newNode
   190  	}
   191  	return true
   192  }

   200  	}

   203  	// go to index node

   265  // PopAt removes an element from the list and return them, the second argument will be 'true'.
   266  // If there is no element on this position the default value will be returned and 'false' as the second argument.
   267  func (l *List[T]) PopAt(index int) (val T, exists bool) {
   268  	if l.root == nil || index < 0 || l.len <= index {
   269  		return val, false
   270  	}
   271  
   272  	l.len -= 1
   273  	e := l.root
   274  	if l.len == 0 {
   275  		l.root = nil
   276  	} else {
   277  		for i := 0; i < index; i++ {
   278  			e = e.next
   279  		}
   280  		e.prev.next = e.next
   281  		e.next.prev = e.prev
   282  		if e == l.root {
   283  			l.root = e.next
   284  		}
   285  	}
   286  	e.clear()
   287  
   288  	return e.data, true
   289  }

   408  func (l *List[T]) MoveAfter(from, to int) bool {

   410  		return false
   411  	}
   412  	if from == to || from-to == 1 {
   413  		return true
   414  	}

   452  func (l *List[T]) Equal(other *List[T]) bool {
   453  	if l.Len() != other.Len() {
   454  		return false
   455  	}
   456  	if l.Len() == 0 {
   457  		return true
   458  	}
   459  
   460  	// Can't use .Index because of O^2
   461  	next, stop := iter.Pull(other.Seq())
   462  	for v := range l.Seq() {
   463  		o, exists := next()
   464  		if !exists {
   465  			stop()
   466  			return false
   467  		}
   468  		if o != v {
   469  			return false
   470  		}
   471  	}
   472  	return true
