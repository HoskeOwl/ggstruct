    74  func (t *Trie[V]) Insert(key string, value V) {
    75  	cur := t.root
    76  	for _, r := range key {
    77  		if n, exists := cur.leafs[r]; exists {
    78  			cur = n
    79  		} else {
    80  			n = newNode(r, cur)
    81  			cur.leafs[r] = n
    82  			cur = n
    83  			t.nodesLen++
    84  		}
    85  	}
    86  	cur.value = &value
    87  	t.valueLen++

   120  // Remove removes node with value if exists. Also remove all empty upper nodes.
   121  func (t *Trie[V]) Remove(key string) (res V, exists bool) {
   122  	if t.valueLen == 0 {
   123  		exists = false
   124  		return
   125  	}
   126  
   127  	cur := t.getNode(key)
   128  	if cur == nil || cur.value == nil {
   129  		exists = false
   130  		return
   131  	}
   132  
   133  	res = *cur.value
   134  	cur.value = nil
   135  	t.valueLen--
   136  	exists = true
   137  
   138  	if len(cur.leafs) > 0 {
   139  		// keep for leafs
   140  		t.valueLen--
   141  		return
   142  	}
   143  
   144  	for {
   145  		rm := cur
   146  		cur = cur.prev
   147  		delete(cur.leafs, rm.key)
   148  		t.nodesLen--
   149  		rm.Clear()
   150  		if cur == t.root || len(cur.leafs) != 0 || cur.value != nil {
   151  			return
   152  		}
   153  	}
   154  }

   156  // Clear removes all elements.
   157  func (t *Trie[V]) Clear() {
   158  	if t.valueLen == 0 {
   159  		return
   160  	}
   161  
   162  	s := stack.New[*node[V]](slices.Collect(maps.Values(t.root.leafs))...)
   163  	for !s.IsEmpty() {
   164  		n, exists := s.Pop()
   165  		if !exists {
   166  			return
   167  		}
   168  		for _, v := range n.leafs {
   169  			s.Push(v)
   170  		}
   171  		n.Clear()
   172  	}
   173  	t.valueLen = 0
   174  	t.nodesLen = 0
   175  }
